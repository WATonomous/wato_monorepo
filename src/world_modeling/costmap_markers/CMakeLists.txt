cmake_minimum_required(VERSION 3.8)
project(costmap_markers)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(geometry_msgs REQUIRED)
find_package(visualization_msgs REQUIRED)

# CostmapMarkersNode executable
add_executable(costmap_markers_node
  src/costmap_markers_node.cpp
)
target_include_directories(costmap_markers_node PRIVATE include)
target_link_libraries(costmap_markers_node
  PRIVATE
    rclcpp::rclcpp
    ${geometry_msgs_TARGETS}
    ${visualization_msgs_TARGETS}
)

# Install targets
install(TARGETS
  costmap_markers_node
  DESTINATION lib/${PROJECT_NAME}
)

install(DIRECTORY config/
  DESTINATION share/${PROJECT_NAME}/config
)

install(DIRECTORY launch/
  DESTINATION share/${PROJECT_NAME}/launch
)

if(BUILD_TESTING)
endif()

ament_package()
